#User: b-it-bots
#PW: area51
#Hostname: youbot-hbrs#   where # is a running number


sudo passwd	# use the same password as above

#Set static IP addresses and subnet mask
# EtherCAT: 1.1.1.1 / 255.255.255.0
# EtherNet pc1: 192.168.#.101   where # is the running youbot number 
# EtherNet pc2: 192.168.#.102   where # is the running youbot number 

#Add hosts
sudo gedit /etc/hosts
# and add the IP from the other youbot pc. E.g. for the youbot-hbrs2-pc1:
192.168.2.102	youbot-hbrs2-pc2


sudo apt-get remove deja-dup brasero brasero-cdrkit brasero-common tomboy ubuntuone-control-panel ubuntuone-client gbrainy gnome-mahjongg gnome-sudoku gnomine libreoffice-* shotwell simple-scan empathy gwibber thunderbird transmission-* banshee usb-creator-gtk onboard

sudo apt-get install openssh-server beep

ssh-keygen
cat ~/.ssh/id_rsa.pub

#Add key to the related github account name

#Add BRSU Mirror to /etc/apt/source.list

sudo apt-get clean
sudo apt-get autoclean
sudo apt-get autoremove
sudo apt-get update
sudo apt-get upgrade

#Disable distribution upgrade in settings of the update manager

cd ~
rm -rf Documents/ Downloads/ examples.desktop Music/ Pictures/ Public/ Templates/ Videos/

#Follow the instructions of the README in https://github.com/b-it-bots/RoboCupAtWork

# Enable color git outputs
git config --global color.branch auto
git config --global color.diff auto
git config --global color.interactive auto
git config --global color.status auto


# setup bashrc
cd ~
git clone SETUP_REPO
cd SETUP_REPO/bashrc
sudo cp youbot-pc#.bashrc /etc/youbot.bashrc	where # is the pc number
cp user.bashrc ~/.bashrc
#adapt all aliases, variables like ROBOT, ROBOT_ENV and so on

#copy udev rules
cd ~
cd SETUP_REPO/udev_rules
sudo cp *.* /etc/udev/rules.d/


#copy ssh
cat ~/.ssh/id_rsa.pub | ssh user@domain.tld 'cat >> ~/.ssh/authorized_keys'
